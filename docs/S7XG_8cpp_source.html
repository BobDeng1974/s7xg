442 gps_message_t S7XG::gpsData() {
443 
444  char * buffer = _sendAndReturn(GPS_GET_DATA);
445  gps_message_t message;
446 
447  // No data yet:
448  // POSITIONING ( 14.8s )
449  // GPS data OK:
450  // DD UTC( 2019/09/02 12:33:34 ) LAT( 41.601215 N ) LONG( 2.622485 E ) POSITIONING( 3.6s )
451 
452  // Tokenize
453  char * tok = strtok(buffer, " ");
454  message.fix = (0 == strcmp(tok, "DD"));
455  tok = strtok(NULL, " "); // UTC( or (
456  if (!tok) return message;
457 
458  if (message.fix) {
459 
460  char * date = strtok(NULL, " "); // date
461  char * time = strtok(NULL, " "); // time
462  sscanf(date, "%d/%d/%d", &message.year, &message.month, &message.day);
463  sscanf(time, "%d:%d:%d", &message.hour, &message.minute, &message.second);
464 
465  strtok(NULL, " "); // )
466  strtok(NULL, " "); // LAT(
467  tok = strtok(NULL, " "); // latitude
468  message.latitude = atof(tok);
469  tok = strtok(NULL, " "); // latitude sign
470  if ('S' == tok[0]) message.latitude = -message.latitude;
471  strtok(NULL, " "); // )
472  strtok(NULL, " "); // LONG(
473  tok = strtok(NULL, " "); // longitude
474  message.longitude = atof(tok);
475  tok = strtok(NULL, " "); // longitude sign
476  if ('W' == tok[0]) message.longitude = -message.longitude;
477  strtok(NULL, " "); // )
478  strtok(NULL, " "); // POSITIONING(
479 
480  }
481 
482  tok = strtok(NULL, " "); // positioning value
483  message.positioning = atof(tok);
484 
485 
486  return message;
487 
488 }

584 uint8_t S7XG::_readLine() {
585 
586  _buffer[0] = 0;
587  uint8_t pointer = 0;
588  uint8_t flag = 0;
589  uint32_t start = millis();
590  uint32_t timeout = _wait_longer ? S7XG_LONG_TIMEOUT : S7XG_SHORT_TIMEOUT;
591  _wait_longer = false;
592 
593  while (millis() - start < timeout) {
594  if (_stream->available()) {
595  uint8_t ch = _stream->read();
596 
597  #if defined(S7XG_DEBUG_SERIAL)
598  if ((31 < ch) && (ch < 127)) {
599  char ch_buff[6];
600  snprintf(ch_buff, sizeof(ch_buff), "%c", ch);
601  S7XG_DEBUG(ch_buff);
602  }
603  #endif
604 
605  if (flag > 2) {
606  if (0x0A == ch) break;
607  _buffer[pointer++] = ch;
608  _buffer[pointer] = 0;
609  if (S7XG_RX_BUFFER_SIZE == pointer - 1) break;
610  } else if (flag == 2) {
611  flag = (' ' == ch) ? flag + 1 : 0;
612  } else {
613  flag = ('>' == ch) ? flag + 1 : 0;
614  }
615  }
616  }
617  S7XG_DEBUG(F("\n"));
618 
619  return pointer;
620 
621 }

650 bool S7XG::_sendAndACK(PGM_P format_P, ...) {
651 
652  char format[strlen_P(format_P) + 1];
653  memcpy_P(format, format_P, sizeof(format));
654 
655  va_list args;
656  va_start(args, format_P);
657  char command[S7XG_TX_BUFFER_SIZE];
658  int len = vsnprintf(command, sizeof(command), format, args);
659  va_end(args);
660 
661  if (len < S7XG_TX_BUFFER_SIZE) {
662  _flush();
663  _send(command);
664  _readLine();
665  return 0 == strcmp(_buffer, "Ok");
666  }
667  return false;
668 
669 }

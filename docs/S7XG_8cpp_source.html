312 gps_message_t S7XG::gpsData() {
313 
314  char * buffer = _sendAndReturn(GPS_GET_DATA);
315 
316  // No data yet:
317  // POSITIONING ( 14.8s )
318  // GPS data OK:
319  // DD UTC( 2017/11/23 06:59:06 ) LAT( 24.988825 N ) LONG( 121.308326 E ) POSITIONING( 2.9s )
320 
321  // Replace spaces with commas
322  char * p = buffer;
323  while (*p != 0) {
324  if (*p == ' ') *p = ',';
325  *p++;
326  }
327 
328  // Store message here
329  gps_message_t message;
330 
331  // Tokenize
332  char * tok = strtok(buffer, ",");
333  message.fix = (0 == strcmp(tok, "DD"));
334  tok = strtok(NULL, ","); // UTC( or (
335 
336  if (message.fix) {
337 
338  tok = strtok(NULL, ","); // timestamp
339  // TODO
340  tok = strtok(NULL, ","); // )
341  tok = strtok(NULL, ","); // LAT(
342  tok = strtok(NULL, ","); // latitude
343  message.latitude = atof(tok);
344  tok = strtok(NULL, ","); // latitude sign
345  if ('S' == tok[0]) message.latitude = -message.latitude;
346  tok = strtok(NULL, ","); // )
347  tok = strtok(NULL, ","); // LONG(
348  tok = strtok(NULL, ","); // longitude
349  message.longitude = atof(tok);
350  tok = strtok(NULL, ","); // longitude sign
351  if ('W' == tok[0]) message.longitude = -message.longitude;
352  tok = strtok(NULL, ","); // )
353  tok = strtok(NULL, ","); // POSITIONING(
354 
355  }
356 
357  tok = strtok(NULL, ","); // positioning value
358  message.positioning = atof(tok);
359 
360  return message;
361

430 uint8_t S7XG::_readLine(uint32_t timeout) {
431 
432  _buffer[0] = 0;
433  uint8_t pointer = 0;
434  uint8_t flag = 0;
435  uint32_t start = millis();
436 
437  char ch_buff[6];
438 
439  while (millis() - start < timeout) {
440  if (_stream->available()) {
441  uint8_t ch = _stream->read();
442 
443  #if defined(S7XG_DEBUG_SERIAL)
444  if ((31 < ch) && (ch < 127)) {
445  snprintf(ch_buff, sizeof(ch_buff), "%c", ch);
446  S7XG_DEBUG(ch_buff);
447  }
448  #endif
449 
450  if (flag > 2) {
451  if (0x0A == ch) break;
452  _buffer[pointer++] = ch;
453  _buffer[pointer] = 0;
454  if (S7XG_RX_BUFFER_SIZE == pointer - 1) break;
455  } else if (flag == 2) {
456  flag = (' ' == ch) ? flag + 1 : 0;
457  } else {
458  flag = ('>' == ch) ? flag + 1 : 0;
459  }
460  }
461  }
462  S7XG_DEBUG(F("\n"));
463 
464  return pointer;
465 
466 }

495 bool S7XG::_sendAndACK(PGM_P format_P, ...) {
496 
497  char format[strlen_P(format_P) + 1];
498  memcpy_P(format, format_P, sizeof(format));
499 
500  va_list args;
501  va_start(args, format_P);
502  char command[S7XG_TX_BUFFER_SIZE];
503  int len = vsnprintf(command, sizeof(command), format, args);
504  va_end(args);
505 
506  if (len < S7XG_TX_BUFFER_SIZE) {
507  _flush();
508  _send(command);
509  _readLine();
510  return 0 == strcmp(_buffer, "Ok");
511  }
512  return false;
513 
514 }
